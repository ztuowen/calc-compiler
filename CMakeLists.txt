# adapted a bit from http://llvm.org/docs/CMake.html#embedding-llvm-in-your-project

cmake_minimum_required(VERSION 2.8.8)
project(CalcCompiler)

find_package(LLVM REQUIRED CONFIG)

message(STATUS "Found LLVM ${LLVM_PACKAGE_VERSION}")
message(STATUS "Using LLVMConfig.cmake in: ${LLVM_DIR}")

set(CMAKE_CXX_FLAGS "-std=c++11 -fno-rtti")

include_directories(include ${LLVM_INCLUDE_DIRS})
add_definitions(${LLVM_DEFINITIONS})

add_executable(calcc
        calcc.cpp
        src/AST.cpp
        src/parser.cpp
        src/asttools/referrer.cpp
        src/asttools/checker.cpp
        src/asttools/dumper.cpp
        src/asttools/compiler.cpp
        )

llvm_map_components_to_libnames(llvm_libs support core irreader)

target_link_libraries(calcc ${llvm_libs})
